description: The TTP223 is a touch pad detector IC that provides a single touch key with low power consumption and a wide operating voltage range (2.0 V to 5.5 V). It supports multiple output modes (CMOS, open‑drain, direct, toggle) and offers fast and low‑power operation, adjustable sensitivity, and optional maximum key‑on duration.

const int touchPin = 2; // Q pin

void setup() {
  pinMode(touchPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  int state = digitalRead(touchPin);
  Serial.println(state == HIGH ? "Touched" : "Not touched");
  delay(100);
}
