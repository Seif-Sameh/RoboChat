      code: //Robu.in
int analogInput = A1;
float vout = 0.0;
float vin = 0.0;
float R1 = 30000.0; // 30kΩ
float R2 = 7500.0;  // 7.5kΩ
int value = 0;
void setup(){
   pinMode(analogInput, INPUT);
   Serial.begin(9600);
   Serial.print("DC VOLTMETER");
}
void loop(){
   // read the value at analog input
   value = analogRead(analogInput);
   vout = (value * 5.0) / 1024.0; // convert ADC value to voltage
   vin = vout / (R2/(R1+R2));  // calculate input voltage
   Serial.print("INPUT V= ");
   Serial.println(vin,2);
   delay(500);
}
      explaination: The sketch reads the analog input, converts the ADC value to a voltage (0–5V), then scales it back to the original input voltage using the known resistor ratio. The result is printed to the serial monitor.
