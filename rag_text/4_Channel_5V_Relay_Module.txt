sample_sketches: // Arduino example
const int relayPins[] = {12, 11, 10, 9};
const int numRelays = sizeof(relayPins) / sizeof(relayPins[0]);
void setup() {
  for (int i = 0; i < numRelays; i++) {
    pinMode(relayPins[i], OUTPUT);
    digitalWrite(relayPins[i], HIGH); // relays off
  }
}
void loop() {
  for (int i = 0; i < numRelays; i++) {
    digitalWrite(relayPins[i], LOW);  // activate
    delay(1000);
    digitalWrite(relayPins[i], HIGH); // deactivate
    delay(2000);
  }
}
