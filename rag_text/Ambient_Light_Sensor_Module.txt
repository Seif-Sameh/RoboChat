      code: #include <Arduino.h>

const int lightPin = 2; // Digital output pin

void setup() {
  pinMode(lightPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  int lightState = digitalRead(lightPin);
  if (lightState == LOW) {
    Serial.println("Light detected");
  } else {
    Serial.println("No light");
  }
  delay(500);
}
      explaination: Reads the digital output of the LDR module. A LOW value indicates light is detected, while HIGH means no light. The code prints the status to the serial monitor.
