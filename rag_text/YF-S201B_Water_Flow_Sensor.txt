  example_code:
    - Item 1:
      code: #include <LiquidCrystal.h>

LiquidCrystal lcd(8, 9, 4, 5, 6, 7);

int input = A0;
float total = 0;

void setup() {
  Serial.begin(9600);
  lcd.begin(16, 2);
  pinMode(input, INPUT);
}

void loop() {
  unsigned long high = pulseIn(input, HIGH);
  unsigned long low = pulseIn(input, LOW);
  unsigned long time = high + low;
  float frequency = 1000000.0 / time;
  float water = frequency / 7.5; // L/min
  float liters = water / 60.0;   // L/hr
  total += liters;
  lcd.clear();
  lcd.setCursor(0,0);
  lcd.print("VOL: "); lcd.print(water, 2); lcd.print(" L/M");
  lcd.setCursor(0,1);
  lcd.print("TOTAL: "); lcd.print(total, 2); lcd.print(" L");
  delay(1000);
}
      explaination: The sketch counts pulses from the sensor, calculates frequency, converts to flow rate (L/min) using the sensor’s calibration (7.5 pulses per L/min), and accumulates total volume. Results are displayed on an LCD.
