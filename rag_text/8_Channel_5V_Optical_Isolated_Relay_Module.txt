description: Low‑level 5V 8‑channel relay interface board with opto‑isolated inputs. Each channel requires 15‑20mA driver current and drives high‑current relays rated for AC250V/10A or DC30V/10A. The module is compatible with Arduino and other microcontrollers, featuring SPDT low‑noise relays, opto‑coupler isolation, and a standard 5V logic interface.
application_examples:

sample_sketches: // Arduino example for 8‑channel relay
int RelayControl1 = 4;
int RelayControl2 = 5;
int RelayControl3 = 6;
int RelayControl4 = 7;
int RelayControl5 = 8;
int RelayControl6 = 9;
int RelayControl7 = 10;
int RelayControl8 = 11;

void setup() {
  Serial.begin(9600);
  pinMode(RelayControl1, OUTPUT);
  pinMode(RelayControl2, OUTPUT);
  pinMode(RelayControl3, OUTPUT);
  pinMode(RelayControl4, OUTPUT);
  pinMode(RelayControl5, OUTPUT);
  pinMode(RelayControl6, OUTPUT);
  pinMode(RelayControl7, OUTPUT);
  pinMode(RelayControl8, OUTPUT);
}

void loop() {
  digitalWrite(RelayControl1, LOW); // activate channel 1
  delay(1000);
  digitalWrite(RelayControl1, HIGH); // deactivate
  delay(1000);
  // repeat for other channels as needed
}
