sample_sketches: #include <Arduino.h>

int IN1 = 4;
#define ON 0
#define OFF 1

void setup() {
  pinMode(IN1, OUTPUT);
  digitalWrite(IN1, OFF);
}

void loop() {
  digitalWrite(IN1, ON);  // turn relay on
  delay(2000);
  digitalWrite(IN1, OFF); // turn relay off
  delay(2000);
}
